194     /**
195      * Create a string from a BigDecimal making sure that it's not longer than
196      * the available space.
197      *
198      * @param decimalValue
199      *            the decimal value to print
200      * @param precision
201      *            the precision to use
202      * @param availableSpace
203      *            the space available for the value
204      * @return the string representing the value.
205      */
206     private static String dblString(BigDecimal decimalValue, int precision, int availableSpace) {
207         BigDecimal decimal = decimalValue;
208         if (precision >= 0) {
209             decimal = decimalValue.setScale(precision, RoundingMode.HALF_UP);
210         }
211         double absInput = Math.abs(decimalValue.doubleValue());
212         if (absInput > 0d && absInput < MAX_DECIMAL_VALUE_TO_USE_PLAIN_STRING) {
213             String value = decimal.toPlainString();
214             if (value.length() < availableSpace) {
215                 return value;
216             }
217         }
218         String value = decimalValue.toString();
219         while (value.length() > availableSpace) {
220             decimal = decimalValue.setScale(decimal.scale() - 1, BigDecimal.ROUND_HALF_UP);
221             value = decimal.toString();
222         }
223         return value;
224     }

264     /**
265      * detect the decimal type of the value, does it fit in a Double/BigInteger
266      * or must it be a BigDecimal to keep the needed precission.
267      *
268      * @param value
269      *            the String value to check.
270      * @return the type to fit the value
271      */
272     private static Class<?> getDecimalNumberType(String value) {
273         BigDecimal bigDecimal = new BigDecimal(value);
274         if (bigDecimal.abs().compareTo(HeaderCard.LONG_MAX_VALUE_AS_BIG_DECIMAL) > 0 && bigDecimal.remainder(BigDecimal.ONE).compareTo(BigDecimal.ZERO) == 0) {
275             return BigInteger.class;
276         } else if (bigDecimal.equals(BigDecimal.valueOf(Double.valueOf(value)))) {
277             return Double.class;
278         } else {
279             return BigDecimal.class;
280         }
281     }

283     private static Class<?> getIntegerNumberType(String value) {
284         int length = value.length();
285         if (value.charAt(0) == '-' || value.charAt(0) == '+') {
286             length--;
287         }
288         if (length <= HeaderCard.MAX_INTEGER_STRING_SIZE) {
289             return Integer.class;
290         } else if (length <= HeaderCard.MAX_LONG_STRING_SIZE) {
291             return Long.class;
292         } else {
293             return BigInteger.class;
294         }
295     }

297     /**
298      * Read exactly one complete fits header line from the input.
299      *
300      * @param dis
301      *            the data input stream to read the line
302      * @return a string of exactly 80 characters
303      * @throws IOException
304      *             if the input stream could not be read
305      * @throws TruncatedFileException
306      *             is there was not a complete line available in the input.
307      */
308     private static String readOneHeaderLine(HeaderCardCountingArrayDataInput dis) throws IOException, TruncatedFileException {
309         byte[] buffer = new byte[FITS_HEADER_CARD_SIZE];
310         int len;
311         int need = FITS_HEADER_CARD_SIZE;
312         try {
313             while (need > 0) {
314                 len = dis.in().read(buffer, FITS_HEADER_CARD_SIZE - need, need);
315                 if (len == 0) {
316                     throw new TruncatedFileException("nothing to read left");
317                 }
318                 need -= len;
319             }
320         } catch (EOFException e) {
321             if (need == FITS_HEADER_CARD_SIZE) {
322                 throw e;
323             }
324             throw new TruncatedFileException(e.getMessage());
325         }
326         dis.cardRead();
327         return AsciiFuncs.asciiString(buffer);
328     }

330     private static int spaceAvailableForValue(String key) {
331         return FITS_HEADER_CARD_SIZE - (Math.max(key.length(), CONTINUE.key().length()) + SPACE_NEEDED_FOR_EQUAL_AND_TWO_BLANKS);
332     }

334     private static ArrayDataInput stringToArrayInputStream(String card) {
335         byte[] bytes = AsciiFuncs.getBytes(card);
336         if (bytes.length % FITS_HEADER_CARD_SIZE != 0) {
337             byte[] newBytes = new byte[bytes.length + FITS_HEADER_CARD_SIZE - bytes.length % FITS_HEADER_CARD_SIZE];
338             System.arraycopy(bytes, 0, newBytes, 0, bytes.length);
339             Arrays.fill(newBytes, bytes.length, newBytes.length, (byte) ' ');
340             bytes = newBytes;
341         }
342         return new BufferedDataInputStream(new ByteArrayInputStream(bytes));
343     }

370     public HeaderCard(HeaderCardCountingArrayDataInput dis) throws TruncatedFileException, IOException {
371         this.key = null;
372         this.value = null;
373         this.comment = null;
374         this.isString = false;
375 
376         String card = readOneHeaderLine(dis);
377 
378         if (FitsFactory.getUseHierarch() && card.startsWith(HIERARCH_WITH_BLANK)) {
379             hierarchCard(card, dis);
380             return;
381         }
382 
383         // extract the key
384         this.key = card.substring(0, MAX_KEYWORD_LENGTH).trim();
385 
386         // if it is an empty key, assume the remainder of the card is a comment
387         if (this.key.isEmpty()) {
388             this.comment = card.substring(MAX_KEYWORD_LENGTH);
389             return;
390         }
391 
392         // Non-key/value pair lines are treated as keyed comments
393         if (this.key.equals(COMMENT.key()) || this.key.equals(HISTORY.key()) || !card.startsWith("=", MAX_KEYWORD_LENGTH)) {
394             this.comment = card.substring(MAX_KEYWORD_LENGTH).trim();
395             return;
396         }
397         extractValueCommentFromString(dis, card);
398     }

601     /**
602      * Create a HeaderCard from its component parts
603      *
604      * @param key
605      *            Keyword (null for a COMMENT)
606      * @param value
607      *            Value
608      * @param comment
609      *            Comment
610      * @param nullable
611      *            Is this a nullable value card?
612      * @exception HeaderCardException
613      *                for any invalid keyword or value
614      */
615     private HeaderCard(String key, String value, String comment, boolean nullable, boolean isString) throws HeaderCardException {
616         this.isString = isString;
617         if (key == null && value != null) {
618             throw new HeaderCardException("Null keyword with non-null value");
619         } else if (key != null && key.length() > HeaderCard.MAX_KEYWORD_LENGTH && //
620                 (!FitsFactory.getUseHierarch() || !key.startsWith(HIERARCH_WITH_DOT))) {
621             throw new HeaderCardException("Keyword too long");
622         }
623         if (value != null) {
624             value = value.replaceAll(" *$", "");
625 
626             if (value.startsWith("'")) {
627                 if (value.charAt(value.length() - 1) != '\'') {
628                     throw new HeaderCardException("Missing end quote in string value");
629                 }
630                 value = value.substring(1, value.length() - 1).trim();
631             }
632             // Remember that quotes get doubled in the value...
633             if (!FitsFactory.isLongStringsEnabled()
634                     && value.replace("'", "''").length() > (this.isString ? HeaderCard.MAX_STRING_VALUE_LENGTH : HeaderCard.MAX_VALUE_LENGTH)) {
635                 throw new HeaderCardException("Value too long");
636             }
637         }
638 
639         this.key = key;
640         this.value = value;
641         this.comment = comment;
642         this.nullable = nullable;
643     }

645     /**
646      * @return the size of the card in blocks of 80 bytes. So normally every
647      *         card will return 1. only long stings can return more than one.
648      */
649     public int cardSize() {
650         if (this.isString && this.value != null && FitsFactory.isLongStringsEnabled()) {
651             int maxStringValueLength = maxStringValueLength();
652             String stringValue = this.value.replace("'", "''");
653             if (stringValue.length() > maxStringValueLength) {
654                 // this is very bad for performance but it is to difficult to
655                 // keep the cardSize and the toString compatible at all times
656                 return toString().length() / FITS_HEADER_CARD_SIZE;
657             }
658         }
659         return 1;
660     }

690     /**
691      * @param clazz
692      *            the requested class of the value
693      * @param defaultValue
694      *            the value if the card was not present.
695      * @param <T>
696      *            the type of the requested class
697      * @return the value from this card as a specific type
698      */
699     public <T> T getValue(Class<T> clazz, T defaultValue) {
700         if (String.class.isAssignableFrom(clazz)) {
701             return clazz.cast(this.value);
702         } else if (this.value == null || this.value.isEmpty()) {
703             return defaultValue;
704         } else if (Boolean.class.isAssignableFrom(clazz)) {
705             return clazz.cast(getBooleanValue((Boolean) defaultValue));
706         }
707         BigDecimal parsedValue;
708         try {
709             parsedValue = new BigDecimal(this.value);
710         } catch (NumberFormatException e) {
711             return defaultValue;
712         }
713         if (Integer.class.isAssignableFrom(clazz)) {
714             return clazz.cast(parsedValue.intValueExact());
715         } else if (Long.class.isAssignableFrom(clazz)) {
716             return clazz.cast(parsedValue.longValueExact());
717         } else if (Double.class.isAssignableFrom(clazz)) {
718             return clazz.cast(parsedValue.doubleValue());
719         } else if (Float.class.isAssignableFrom(clazz)) {
720             return clazz.cast(parsedValue.floatValue());
721         } else if (BigDecimal.class.isAssignableFrom(clazz)) {
722             return clazz.cast(parsedValue);
723         } else if (BigInteger.class.isAssignableFrom(clazz)) {
724             return clazz.cast(parsedValue.toBigIntegerExact());
725         } else {
726             throw new IllegalArgumentException("unsupported class " + clazz);
727         }
728     }

877     /**
878      * Same as {@link #toString()} just with a prefetched settings object
879      * 
880      * @param settings
881      *            the settings to use for writing the header card
882      * @return the string representing the card.
883      */
884     protected String toString(final FitsSettings settings) {
885         int alignSmallString = NORMAL_SMALL_STRING_ALIGN_POSITION;
886         int alignPosition = NORMAL_ALIGN_POSITION;
887         FitsLineAppender buf = new FitsLineAppender();
888         // start with the keyword, if there is one
889         if (this.key != null) {
890             if (this.key.length() > HIERARCH_WITH_BLANK_LENGTH && this.key.startsWith(HIERARCH_WITH_DOT)) {
891                 settings.getHierarchKeyFormatter().append(this.key, buf);
892                 alignSmallString = buf.length();
893                 alignPosition = buf.length();
894             } else {
895                 buf.append(this.key);
896                 if (this.key.isEmpty()) {
897                     buf.append(' ');
898                 }
899                 buf.appendSpacesTo(MAX_KEYWORD_LENGTH);
900             }
901         }
902         FitsSubString commentSubString = new FitsSubString(this.comment);
903         if (FITS_HEADER_CARD_SIZE - alignPosition - MAX_LONG_STRING_CONTINUE_OVERHEAD < commentSubString.length()) {
904             // with alignment the comment would not fit so lets make more space
905             alignPosition = Math.max(buf.length(), FITS_HEADER_CARD_SIZE - MAX_LONG_STRING_CONTINUE_OVERHEAD - commentSubString.length());
906             alignSmallString = buf.length();
907         }
908         boolean commentHandled = false;
909         if (this.value != null || this.nullable) {
910             if (settings.isSkipBlankAfterAssign()) {
911                 buf.append('=');
912             } else {
913                 buf.append("= ");
914             }
915             if (this.value != null) {
916 
917                 if (this.isString) {
918                     commentHandled = stringValueToString(alignSmallString, alignPosition, buf, commentHandled);
919                 } else {
920                     buf.appendSpacesTo(alignPosition - this.value.length());
921                     buf.append(this.value);
922                 }
923             } else {
924                 // Pad out a null value.
925                 buf.appendSpacesTo(alignPosition);
926             }
927             // is there space left for a comment?
928             int spaceLeft = FITS_HEADER_CARD_SIZE - buf.length();
929             int spaceLeftInCard = spaceLeft % FITS_HEADER_CARD_SIZE;
930             commentSubString.getAdjustedLength(spaceLeftInCard - MAX_LONG_STRING_CONTINUE_OVERHEAD);
931             // if there is a comment, add a comment delimiter
932             if (!commentHandled && commentSubString.length() > 0) {
933                 buf.append(" / ");
934             }
935         } else if (commentSubString.startsWith("=")) {
936             buf.append("  ");
937         }
938 
939         // finally, add any comment
940         if (!commentHandled && commentSubString.length() > 0) {
941             if (commentSubString.startsWith(" ")) {
942                 commentSubString.skip(1);
943             }
944             // is there space left for a comment?
945             commentSubString.getAdjustedLength((FITS_HEADER_CARD_SIZE - buf.length()) % FITS_HEADER_CARD_SIZE);
946             buf.append(commentSubString);
947             commentSubString.rest();
948             if (commentSubString.length() > 0) {
949                 LOG.log(Level.INFO, "" + this.key + " was trimmed to fit");
950             }
951         }
952         buf.completeLine();
953         return buf.toString();
954     }

956     /**
957      * @return the type of the value.
958      */
959     public Class<?> valueType() {
960         if (this.isString) {
961             return String.class;
962         } else if (this.value != null) {
963             String trimedValue = this.value.trim();
964             if ("T".equals(trimedValue) || "F".equals(trimedValue)) {
965                 return Boolean.class;
966             } else if (HeaderCard.LONG_REGEX.matcher(trimedValue).matches()) {
967                 return getIntegerNumberType(trimedValue);
968             } else if (HeaderCard.IEEE_REGEX.matcher(trimedValue).find()) {
969                 return getDecimalNumberType(trimedValue);
970             }
971         }
972         return null;
973     }

976         // extract the value/comment part of the string
977         ParsedValue parsedValue = FitsHeaderCardParser.parseCardValue(card);
978 
979         if (FitsFactory.isLongStringsEnabled() && parsedValue.isString() && parsedValue.getValue().endsWith("&")) {
980             longStringCard(dis, parsedValue);
981         } else {
982             this.value = parsedValue.getValue();

985             if (!this.isString && this.value.indexOf('\'') >= 0) {
986                 throw new IllegalArgumentException("no single quotes allowed in values");
987             }

1012     private void longStringCard(HeaderCardCountingArrayDataInput dis, ParsedValue parsedValue) throws IOException, TruncatedFileException {
1013         // ok this is a longString now read over all continues.
1014         StringBuilder longValue = new StringBuilder();
1015         StringBuilder longComment = null;
1016         ParsedValue continueCard = parsedValue;
1017         do {
1018             if (continueCard.getValue() != null) {
1019                 longValue.append(continueCard.getValue());
1020             }
1021             if (continueCard.getComment() != null) {
1022                 if (longComment == null) {
1023                     longComment = new StringBuilder();
1024                 } else {
1025                     longComment.append(' ');
1026                 }
1027                 longComment.append(continueCard.getComment());
1028             }
1029             continueCard = null;
1030             if (longValue.length() > 0 && longValue.charAt(longValue.length() - 1) == '&') {
1031                 longValue.setLength(longValue.length() - 1);
1032                 dis.mark();
1033                 String card = readOneHeaderLine(dis);
1034                 if (card.startsWith(CONTINUE.key())) {
1035                     // extract the value/comment part of the string
1036                     continueCard = FitsHeaderCardParser.parseCardValue(card);
1037                 } else {
1038                     // the & was part of the string put it back.
1039                     longValue.append('&');
1040                     // ok move the input stream one card back.
1041                     dis.reset();
1042                 }
1043             }
1044         } while (continueCard != null);
1045         this.comment = longComment == null ? null : longComment.toString();
1046         this.value = longValue.toString();
1047         this.isString = true;
1048     }

1058     private boolean stringValueToString(int alignSmallString, int alignPosition, FitsLineAppender buf, boolean commentHandled) {
1059         String stringValue = this.value.replace("'", "''");
1060         if (FitsFactory.isLongStringsEnabled() && stringValue.length() > maxStringValueLength()) {
1061             writeLongStringValue(buf, stringValue);
1062             commentHandled = true;
1063         } else {
1064             // left justify the string inside the quotes
1065             buf.append('\'');
1066             buf.append(stringValue);
1067             buf.appendSpacesTo(alignSmallString);
1068             buf.append('\'');
1069             // Now add space to the comment area starting at column
1070             // 30
1071             buf.appendSpacesTo(alignPosition);
1072         }
1073         return commentHandled;
1074     }

1076     private void writeLongStringValue(FitsLineAppender buf, String stringValueString) {
1077         FitsSubString stringValue = new FitsSubString(stringValueString);
1078         FitsSubString commentValue = new FitsSubString(this.comment);
1079         // We assume that we've made the test so that
1080         // we need to write a long string.
1081         // We also need to be careful that single quotes don't
1082         // make the string too long and that we don't split
1083         // in the middle of a quote.
1084         stringValue.getAdjustedLength(FITS_HEADER_CARD_SIZE - buf.length() - MAX_LONG_STRING_CONTINUE_OVERHEAD);
1085         // No comment here since we're using as much of the card
1086         // as we can
1087         buf.append('\'');
1088         buf.append(stringValue);
1089         buf.append("&'");
1090         buf.completeLine();
1091         stringValue.rest();
1092         if (commentValue.startsWith(" ")) {
1093             commentValue.skip(1);
1094         }
1095         while (stringValue.length() > 0) {
1096             stringValue.getAdjustedLength(MAX_LONG_STRING_VALUE_LENGTH);
1097             if (stringValue.fullLength() > MAX_LONG_STRING_VALUE_LENGTH) {
1098                 buf.append(CONTINUE_CARD_PREFIX);
1099                 buf.append(stringValue);
1100                 buf.append("&'");
1101                 stringValue.rest();
1102             } else {
1103                 if (commentValue.length() > MAX_LONG_STRING_VALUE_WITH_COMMENT_LENGTH - stringValue.length()) {
1104                     // ok comment does not fit lets give it a little more room
1105                     stringValue.getAdjustedLength(STRING_SPLIT_POSITION_FOR_EXTRA_COMMENT_SPACE);
1106                     if (stringValue.fullLength() > stringValue.length()) {
1107                         buf.append(CONTINUE_CARD_PREFIX);
1108                         buf.append(stringValue);
1109                         buf.append("&'");
1110                     } else {
1111                         buf.append(CONTINUE_CARD_PREFIX);
1112                         buf.append(stringValue);
1113                         buf.append("'");
1114                     }
1115                     int spaceForComment = buf.spaceLeftInLine() - MAX_LONG_STRING_CONTINUE_OVERHEAD;
1116                     commentValue.getAdjustedLength(spaceForComment);
1117                 } else {
1118                     buf.append(CONTINUE_CARD_PREFIX);
1119                     buf.append(stringValue);
1120                     buf.append('\'');
1121                 }
1122                 if (commentValue.length() > 0) {
1123                     buf.append(" / ");
1124                     buf.append(commentValue);
1125                     commentValue.rest();
1126                 }
1127                 buf.completeLine();
1128                 stringValue.rest();
1129             }
1130         }
1131     }
