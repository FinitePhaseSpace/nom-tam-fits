45  /**
46   * This class is intended for high performance I/O in scientific applications.
47   * It combines the functionality of the BufferedInputStream and the
48   * DataInputStream as well as more efficient handling of arrays. This minimizes
49   * the number of method calls that are required to read data. Informal tests of
50   * this method show that it can be as much as 10 times faster than using a
51   * DataInputStream layered on a BufferedInputStream for writing large arrays.
52   * The performance gain on scalars or small arrays will be less but there should
53   * probably never be substantial degradation of performance.
54   * <p>
55   * Many new read calls are added to allow efficient reading off array data. The
56   * read(Object o) call provides for reading a primitive array of arbitrary type
57   * or dimensionality. There are also reads for each type of one dimensional
58   * array.
59   * <p>
60   * Note that there is substantial duplication of code to minimize method
61   * invocations. E.g., the floating point read routines read the data as integer
62   * values and then convert to float. However the integer code is duplicated
63   * rather than invoked. There has been considerable effort expended to ensure
64   * that these routines are efficient, but they could easily be superceded if an
65   * efficient underlying I/O package were ever delivered as part of the basic
66   * Java libraries. [This has subsequently happened with the NIO package and in
67   * an ideal universe these classes would be rewritten to take advantage of NIO.]
68   * <p>
69   * Testing and timing routines are provided in the
70   * nom.tam.util.test.BufferedFileTester class. Version 1.1: October 12, 2000:
71   * Fixed handling of EOF to return partially read arrays when EOF is detected.
72   * Version 1.2: July 20, 2009: Added handling of very large Object arrays.
73   * Additional work is required to handle very large arrays generally.
74   */
75  public class BufferedDataInputStream extends BufferedInputStream implements ArrayDataInput {

139     /**
140      * For array reads return an EOF if unable to read any data.
141      * 
142      * @param e
143      *            the eof exception that happened.
144      * @param i
145      *            the current index
146      * @param start
147      *            the start index
148      * @param length
149      *            the element length
150      * @return the number of bytes read before the end of file exception.
151      * @throws EOFException
152      *             if no extra bytes could be read
153      */
154     private int eofCheck(EOFException e, int i, int start, int length) throws EOFException {
155 
156         if (i == start) {
157             throw e;
158         } else {
159             return (i - start) * length;
160         }
161     }

173     @Override
174     public int read(byte[] obuf, int offset, int length) throws IOException {
175         int total = 0;
176         int remainingToRead = length;
177         int currentOffset = offset;
178         while (remainingToRead > 0) {
179             // Use just the buffered I/O to get needed info.
180             int xlen = super.read(obuf, currentOffset, remainingToRead);
181             if (xlen <= 0) {
182                 if (total == 0) {
183                     throw new EOFException();
184                 } else {
185                     return total;
186                 }
187             } else {
188                 remainingToRead -= xlen;
189                 total += xlen;
190                 currentOffset += xlen;
191             }
192         }
193         return total;
194     }

314     /**
315      * Emulate the deprecated DataInputStream.readLine() method. Originally we
316      * used the method itself, but Alan Brighton suggested using a
317      * java.io.BufferedReader to eliminate the deprecation warning. This was
318      * used for a long time, but more recently we noted that this doesn't work.
319      * We now use a simple method that largely ignores character encodings and
320      * only uses the "\n" as the line separator. This method is slow regardless.
321      * In the current version
322      * 
323      * @return The String read.
324      * @deprecated Use {@link java.io.BufferedReader} methods.
325      */
326     @Deprecated
327     @Override
328     public String readLine() throws IOException {
329         // Punt on this and use BufferedReader routines.
330         StringBuilder b = new StringBuilder(0);
331         int chr;
332         while ((chr = read()) >= 0) {
333             if (chr != '\n') {
334                 b.append((char) chr);
335             } else {
336                 return b.toString();
337             }
338         }
339         return b.toString();
340     }

392     @Override
393     public void skipAllBytes(long toSkip) throws IOException {
394         long need = toSkip;
395         while (need > 0) {
396             try {
397                 long got = skip(need);
398                 if (got > 0) {
399                     need -= got;
400                 } else {
401                     break;
402                 }
403             } catch (IOException e) {
404                 long lastNeed = need;
405                 need = handleExceptionInSkip(need, e);
406                 if (need >= lastNeed) {
407                     break;
408                 }
409             }
410         }
411         if (need > 0) {
412             throw new EOFException();
413         }
414     }
415 
416     private long handleExceptionInSkip(long skip, IOException e) throws IOException {
417         LOG.log(Level.WARNING, "Error while skipping bytes", e);
418         // Some input streams (process outputs) don't allow
419         // skipping. The kludgy solution here is to
420         // try to do a read when we get an error in the skip....
421         // Real IO errors will presumably cause an error
422         // in these reads too.
423         if (this.skipBuf == null) {
424             this.skipBuf = new byte[BufferedDataInputStream.SKIP_BUFFER_SIZE];
425         }
426         long remainingToSkip = skip;
427         while (remainingToSkip > BufferedDataInputStream.SKIP_BUFFER_SIZE) {
428             int got = read(this.skipBuf, 0, BufferedDataInputStream.SKIP_BUFFER_SIZE);
429             if (got <= 0) {
430                 break;
431             }
432             remainingToSkip -= got;
433         }
434         while (remainingToSkip > 0) {
435             int got = read(this.skipBuf, 0, (int) remainingToSkip);
436             if (got <= 0) {
437                 break;
438             }
439             remainingToSkip -= got;
440         }
441         return remainingToSkip;
442     }
