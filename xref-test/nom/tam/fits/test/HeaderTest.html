240         FitsFactory.setLongStringsEnabled(false);
241         String seq = "0123456789";
242         String lng = "";
243         String sixty = seq + seq + seq + seq + seq + seq;
244 
245         for (int i = 0; i < 20; i += 1) {
246             lng += seq;
247         }
248         Fits f = null;
249         try {
250             f = new Fits("target/ht1.fits");
251             Header hdr = f.getHDU(0).getHeader();
252             assertEquals("Initial state:", false, FitsFactory.isLongStringsEnabled());
253             FitsFactory.setLongStringsEnabled(true);
254             assertEquals("Set state:", true, FitsFactory.isLongStringsEnabled());
255             hdr.addValue("LONG1", lng, "Here is a comment that is also very long and will be truncated at least a little");
256             hdr.addValue("LONG2", "xx'yy'zz" + lng, "Another comment");
257             hdr.addValue("LONG3", "xx'yy'zz" + lng, null);
258             hdr.addValue("SHORT", "A STRING ENDING IN A &", null);
259             hdr.addValue("LONGISH", lng + "&", null);
260             hdr.addValue("LONGSTRN", "OGIP 1.0", "Uses long strings");
261 
262             hdr.addValue("APOS1", sixty + "''''''''''", "Should be 70 chars long");
263             hdr.addValue("APOS2", sixty + " ''''''''''", "Should be 71 chars long");
264 
265             // Now try to read the values back.
266             BufferedFile bf = null;
267             try {
268                 bf = new BufferedFile("target/ht4.hdr", "rw");
269                 hdr.write(bf);
270             } finally {
271                 SaveClose.close(bf);
272             }
273             String val = hdr.getStringValue("LONG1");
274             assertEquals("LongT1", val, lng);
275             val = hdr.getStringValue("LONG2");
276             assertEquals("LongT2", val, "xx'yy'zz" + lng);
277             assertEquals("APOS1", hdr.getStringValue("APOS1").length(), 70);
278             assertEquals("APOS2", hdr.getStringValue("APOS2").length(), 71);
279 
280             // long3 should not have a comment!
281             assertNull(hdr.findCard("LONG3").getComment());
282 
283             String string = hdr.findCard("LONG1").toString();
284             val = FitsHeaderCardParser.parseCardValue(string).getValue();
285             FitsFactory.setLongStringsEnabled(false);
286             val = FitsHeaderCardParser.parseCardValue(string).getValue();
287             FitsFactory.setLongStringsEnabled(true);
288 
289             assertEquals("LongT3", true, !val.equals(lng));
290             assertEquals("Longt4", true, val.length() <= 70);
291             assertEquals("longamp1", hdr.getStringValue("SHORT"), "A STRING ENDING IN A &");
292             try {
293                 bf = new BufferedFile("target/ht4.hdr", "r");
294                 hdr = new Header(bf);
295                 assertEquals("Set state2:", true, FitsFactory.isLongStringsEnabled());
296                 val = hdr.getStringValue("LONG1");
297                 assertEquals("LongT5", val, lng);
298                 val = hdr.getStringValue("LONG2");
299                 assertEquals("LongT6", val, "xx'yy'zz" + lng);
300                 assertEquals("longamp2", hdr.getStringValue("LONGISH"), lng + "&");
301                 assertEquals("APOS1b", hdr.getStringValue("APOS1").length(), 70);
302                 assertEquals("APOS2b", hdr.getStringValue("APOS2").length(), 71);
303                 assertEquals("APOS2c", hdr.getStringValue("APOS2"), sixty + " ''''''''''");
304                 assertEquals("longamp1b", hdr.getStringValue("SHORT"), "A STRING ENDING IN A &");
305                 assertEquals("longamp2b", hdr.getStringValue("LONGISH"), lng + "&");
306 
307                 int cnt = hdr.getNumberOfCards();
308                 int pcnt = hdr.getNumberOfPhysicalCards();
309                 // This should remove all three cards associated with
310                 // LONG1
311                 hdr.removeCard("LONG1");
312                 assertEquals("deltest", cnt - 1, hdr.getNumberOfCards());

315                 hdr.removeCard("LONG2");

396     /**
397      * split in own method, for debugging (drop to frame)
398      * 
399      * @param cardValue
400      * @param cardComment
401      * @throws Exception
402      */
403     private void checkOneCombination(String cardValue, String cardComment) throws Exception {
404         try {
405             HeaderCard headerCard = new HeaderCard("CARD", cardValue, cardComment);
406             String cardString = headerCard.toString();
407             assertEquals(cardString.length(), headerCard.cardSize() * 80);
408             byte[] bytes = new byte[cardString.length() + 160];
409             Arrays.fill(bytes, (byte) ' ');
410             System.arraycopy(AsciiFuncs.getBytes(cardString), 0, bytes, 0, cardString.length());
411             HeaderCard rereadCard = new HeaderCard(new BufferedDataInputStream(new ByteArrayInputStream(bytes)));
412 
413             assertEquals(cardValue, rereadCard.getValue());
414             assertEquals(headerCard.getValue(), rereadCard.getValue());
415             assertEquals(cardComment, headerCard.getComment());
416             assertTrue(cardComment.startsWith(rereadCard.getComment() == null ? "" : rereadCard.getComment().replaceAll(" ", "")));
417         } catch (HeaderCardException e) {
418             assertTrue(!FitsFactory.isLongStringsEnabled() && cardValue.length() > 68);
419         }
420     }
