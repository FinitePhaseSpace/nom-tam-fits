103         assertNull("t10", p.getValue());
104         assertEquals("t11", lng, p.getComment());

108             //
109             p = HeaderCard.create("VALUE   = '   ");

135         boolean thrown = false;
136         try {
137             p = new HeaderCard("LONGKEYWORD", 123, "COMMENT");
138         } catch (Exception e) {
139             thrown = true;
140         }
141         assertEquals("x5", true, thrown);
142 
143         thrown = false;
144         String lng = "00000000001111111111222222222233333333334444444444555555555566666666667777777777";
145         try {
146             Header.setLongStringsEnabled(false);
147             p = new HeaderCard("KEY", lng, "COMMENT");
148         } catch (Exception e) {
149             thrown = true;
150         }
151         assertEquals("x6", true, thrown);
152 
153         // Only trailing spaces are stripped.
154         p = new HeaderCard("STRING", "VALUE", null);
155         assertEquals("x6", "VALUE", p.getValue());
156 
157         p = new HeaderCard("STRING", "VALUE ", null);
158         assertEquals("x7", "VALUE", p.getValue());
159 
160         p = new HeaderCard("STRING", " VALUE", null);
161         assertEquals("x8", " VALUE", p.getValue());
162 
163         p = new HeaderCard("STRING", " VALUE ", null);
164         assertEquals("x9", " VALUE", p.getValue());
165 
166         p = new HeaderCard("QUOTES", "ABC'DEF", null);
167         assertEquals("x10", "ABC'DEF", p.getValue());
168         assertEquals("x10b", p.toString().indexOf("''") > 0, true);
169 
170         p = new HeaderCard("QUOTES", "ABC''DEF", null);
171         assertEquals("x11", "ABC''DEF", p.getValue());
172         assertEquals("x10b", p.toString().indexOf("''''") > 0, true);

473     @Test
474     public void testKeyWordNullability() throws Exception {
475         assertEquals("TEST    = 'VALUE   '           / COMMENT                                        ", new HeaderCard("TEST", "VALUE", "COMMENT", true).toString());
476         assertEquals("TEST    = 'VALUE   '           / COMMENT                                        ", new HeaderCard("TEST", "VALUE", "COMMENT", false).toString());
477         assertEquals("TEST    =                      / COMMENT                                        ", new HeaderCard("TEST", null, "COMMENT", true).toString());
478         assertEquals("TEST    COMMENT                                                                 ", new HeaderCard("TEST", null, "COMMENT", false).toString());
479         HeaderCardException actual = null;
480         try {
481             new HeaderCard(null, "VALUE", "COMMENT", true);
482         } catch (HeaderCardException e) {
483             actual = e;
484         }
485         Assert.assertNotNull(actual);
486         assertEquals(true, new HeaderCard("TEST", "VALUE", "COMMENT", true).isKeyValuePair());
487         assertEquals(true, new HeaderCard("TEST", "VALUE", "COMMENT", false).isKeyValuePair());
488         assertEquals(false, new HeaderCard("TEST", null, "COMMENT", true).isKeyValuePair());
489         assertEquals(false, new HeaderCard("TEST", null, "COMMENT", false).isKeyValuePair());
490
