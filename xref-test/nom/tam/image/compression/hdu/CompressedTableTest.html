50  public class CompressedTableTest {
51  
52      private final double[][][] doubles = new double[50][5][5];
53  
54      public CompressedTableTest() {
55          super();
56          Random random = new Random();
57          for (int i1 = 0; i1 < this.doubles.length; i1++) {
58              for (int i2 = 0; i2 < this.doubles[i1].length; i2++) {
59                  for (int i3 = 0; i3 < this.doubles[i1][i2].length; i3++) {
60                      this.doubles[i1][i2][i3] = random.nextDouble();
61                  }
62              }
63          }
64      }
65  
66      @Test
67      public void testBasicBinaryTable() throws Exception {
68  
69          try {
70              System.setProperty("compressed.table.experimental", "true");
71  
72              Fits fitsComp = new Fits("src/test/resources/nom/tam/table/comp/testBinaryTable.fits.fz");
73              CompressedTableHDU cfitsioTable = (CompressedTableHDU) fitsComp.getHDU(1);
74  
75              cfitsioTable.info(System.out);
76  
77              BinaryTableHDU binaryTable2HDU = cfitsioTable.asBinaryTableHDU();
78  
79              Fits fits = new Fits();
80              fits.addHDU(binaryTable2HDU);
81              fits.write(new File("target/testBinaryTable_uncompressed.fits"));
82              fits.close();
83  
84              CompressedTableHDU compressed = CompressedTableHDU.fromBinaryTableHDU(binaryTable2HDU, 10).compress();
85              compressed.compress();
86              fits = new Fits();
87              fits.addHDU(compressed);
88              fits.write(new File("target/testBinaryTable_recompressed.fits"));
89              fits.close();
90  
91          } finally {
92              System.setProperty("compressed.table.experimental", "false");
93          }
94  
95          Fits fits = new Fits("target/testBinaryTable_uncompressed.fits");
96          Header header = fits.getHDU(1).getHeader();
97          Cursor<String, HeaderCard> iter = header.iterator();
98          assertStringCard(Standard.XTENSION, XTENSION_BINTABLE, iter.next());
99          assertIntCard(Standard.BITPIX, 8, iter.next());
100         assertIntCard(Standard.NAXIS, 2, iter.next());
101         assertIntCard(Standard.NAXISn.n(1), 200, iter.next());
102         assertIntCard(Standard.NAXISn.n(2), 50, iter.next());
103         assertIntCard(Standard.PCOUNT, 0, iter.next());
104         assertIntCard(Standard.GCOUNT, 1, iter.next());
105         assertIntCard(Standard.TFIELDS, 1, iter.next());
106         // the order of the next two fields is not fix
107         assertStringCard(Standard.TFORMn.n(1), "25D", header.card(Standard.TFORMn.n(1)).card());
108         assertStringCard(Standard.TDIMn.n(1), "(5,5)", header.card(Standard.TDIMn.n(1)).card());
109         fits.close();
110 
111         fits = new Fits("target/testBinaryTable_recompressed.fits");
112         header = fits.getHDU(1).getHeader();
113         iter = header.iterator();
114         assertStringCard(Standard.XTENSION, XTENSION_BINTABLE, iter.next());
115         assertIntCard(Standard.BITPIX, 8, iter.next());
116         assertIntCard(Standard.NAXIS, 2, iter.next());
117         assertIntCard(Standard.NAXISn.n(1), 16, iter.next());
118         assertIntCard(Standard.NAXISn.n(2), 5, iter.next());
119         assertIntCard(Standard.PCOUNT, 1380, iter.next());
120         assertIntCard(Standard.GCOUNT, 1, iter.next());
121         assertIntCard(Standard.TFIELDS, 1, iter.next());
122         // the order of the next two fields is not fix
123         assertStringCard(Standard.TFORMn.n(1), "1QB", header.card(Standard.TFORMn.n(1)).card());
124         assertStringCard(Standard.TDIMn.n(1), "(5,5)", header.card(Standard.TDIMn.n(1)).card());
125         fits.close();
126     }
127 
128     private void assertIntCard(IFitsHeader expectedKey, int expectedValue, HeaderCard card) {
129         Assert.assertEquals(expectedKey.key(), card.getKey());
130         Assert.assertEquals(Integer.valueOf(expectedValue), card.getValue(Integer.class, -1));
131     }
132 
133     private void assertStringCard(IFitsHeader expectedKey, String expectedValue, HeaderCard card) {
134         Assert.assertEquals(expectedKey.key(), card.getKey());
135         Assert.assertEquals(expectedValue, card.getValue());
136     }
